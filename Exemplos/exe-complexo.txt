# Função somar
def somar(a, b):
   x = a + b
   return x

# Testando função em uma lista
lista = [1, 2, 3, 4]
resultado = 0
indice = 0
while(indice < len(lista)):
   resultado += somar(lista[indice], lista[indice+1])
   indice += 2

# Mostrando resultado
print(resultado)
